00045  * Simple algorithm that computes the count of components of

00067 template <class TMesh, class TMesh_Traits = mesh_traits<TMesh>>
00068 int compute_components(TMesh& m)
00069 {

00071    typedef typename TMesh_Traits::vertex_descriptor vertex_descriptor;
00072    typedef typename TMesh_Traits::vertex_iterator vertex_iterator;
00073    typedef typename TMesh_Traits::vv_iterator vv_iterator;
00074 
00075    boost::function_requires<ComputeComponentsConcept<TMesh, TMesh_Traits> >();
00076 
00077    int component_count = 0;
00078 
00079    std::pair<vertex_iterator,vertex_iterator> vertex_pair = get_all_vertices(m);
00080    std::vector<vertex_descriptor> vertices;
00081 
00082    for (vertex_iterator i = vertex_pair.first;  i != vertex_pair.second; ++i) {
00083       vertices.push_back(*i);
00084    }
00085    std::vector<vertex_descriptor> visited_vertices;
00086 
00087    while (!vertices.empty())
00088    {
00089 
00090 
00091       std::deque<vertex_descriptor> vertex_queue;
00092       vertex_queue.push_back(*(vertices.begin()));
00093       visited_vertices.push_back(*(vertices.begin()));
00094 
00095       while (!vertex_queue.empty())
00096       {
00097          vertex_descriptor vd = *(vertex_queue.begin());
00098          vertex_queue.pop_front();
00099 
00100          if (is_isolated(m, vd)) continue;
00101 
00102          std::pair<vv_iterator, vv_iterator> adjacent_vertices = get_adjacent_vertices(m, vd);
00103 
00104          for (vv_iterator i = adjacent_vertices.first ; i!= adjacent_vertices.second; ++i)
00105          {
00106             vertex_descriptor adjacent_vertex = *i;
00107             if (std::find(visited_vertices.begin(), visited_vertices.end(), adjacent_vertex) != visited_vertices.end())
00108                continue;
00109 
00110             vertex_queue.push_front(adjacent_vertex);
00111             visited_vertices.push_back(adjacent_vertex);
00112          }
00113       }
00114 
00115       std::sort(vertices.begin(), vertices.end());
00116       std::sort(visited_vertices.begin(), visited_vertices.end());
00117 
00118       std::vector<vertex_descriptor> semi_result;
00119 
00120       std::set_difference(vertices.begin(), vertices.end(), visited_vertices.begin(), visited_vertices.end(), std::inserter(semi_result, semi_result.end()));
00121       vertices = semi_result;
00122       visited_vertices.clear();
00123 
00124       ++component_count;
00125    }
